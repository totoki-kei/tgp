# 文字コードはANSI(コメント部分は不問)
# シャープ始まりはコメント(行途中からのコメントには非対応)
# パラメータはスラッシュ区切り、パラメータの内部はスペース区切り
# 頂点のデフォルト値を設定する
# V/<デフォルト頂点座標xyzw>/<デフォルト頂点カラーrgba>/<デフォルトEmitカラーrgba>
# 各パラメータは4個の数値
V/ 0 0 0 1/ 1 1 1 0.5/ 1 1 1 1
# 頂点を追加する
# v/<頂点座標xyzw>/<頂点カラーrgba>/<Emitカラーrgba>
# 各パラメータは0〜4個の数値
# 数値が3個以下の場合は、残りの部分がデフォルト設定値から補われる
v/-1  1  0 // 0.5 0.5 0
v/-1  1 -1 // 1 1 1 0
v/ 0  1 -1 // 0 0.5 0.5
v/ 1  1  0 // 0.5 0.5 0

v/ 0  1  1 // 0 0.5 0.5
v/ 1  1  1 // 1 1 1 0
v/-1  1  1 // 1 1 1 0
v/ 1  0  1 // 0.5 0 0.5
v/-1  0  1 // 0.5 0 0.5
v/-1 -1  1 // 1 1 1 0
v/ 0 -1  1 // 0 0.5 0.5
v/ 1 -1  1 // 1 1 1 0
v/ 1 -1 -1 // 1 1 1 0
v/-1 -1 -1 // 1 1 1 0
v/-1 -1  0 // 0.5 0.5 0
v/-1  0 -1 // 0.5 0 0.5
v/ 1 -1  0 // 0.5 0.5 0
v/ 0 -1 -1 // 0 0.5 0.5
v/ 1  0 -1 // 0.5 0 0.5
# グループ(サブセット)の開始
# グループは、次のグループが開始されるまで継続される
# g/<カラーリングタイプ>/<ベースカラーrgba>/<エフェクトパラメータ>
# カラーリングタイプはN(Normal),E(Emit)から選択(今後追加予定)
# ベースカラーは数値4個、エフェクトパラメータは数値2個
g/L/1 1 1 1/1 0 1 0
